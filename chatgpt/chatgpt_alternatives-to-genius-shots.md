## 2) Silent Session Refresh + Draft Autosave

**`lib/auth/session-watch.tsx`**
```tsx
"use client";
import { useAuth, useClerk } from "@clerk/nextjs";
import { useEffect, useRef } from "react";
import { saveDraftSnapshot } from "@/lib/drafts";

export function SessionWatch({ onExpire }: { onExpire?: () => void }) {
  const { sessionId, isSignedIn, getToken } = useAuth();
  const { signOut } = useClerk();
  const triedRefresh = useRef(false);

  useEffect(() => {
    let alive = true;

    async function check() {
      try {
        // Attempt background token fetch (forces refresh if needed)
        await getToken({ template: "supabase" });
      } catch {
        if (!alive) return;

        // Last-ditch: autosave UI draft before sign-out
        try { await saveDraftSnapshot(); } catch {}

        if (onExpire) onExpire();

        // Hard sign-out to clear stale state
        await signOut({ redirectUrl: `/sign-in?returnTo=${encodeURIComponent(window.location.pathname + window.location.search)}` });
      }
    }

    if (isSignedIn && sessionId && !triedRefresh.current) {
      triedRefresh.current = true;
      void check();
    }
    return () => { alive = false; };
  }, [isSignedIn, sessionId, getToken, signOut, onExpire]);

  return null;
}
```

**Usage in protected pages**
```tsx
// app/(protected)/dashboard/page.tsx

export default function Dashboard() {
  return (
    <>
      <SessionWatch onExpire={() => {
```

## 3) Supabase client (Clerk token)

**`lib/supabase/browser-client.ts`**
```ts
"use client";
import { createBrowserClient } from "@supabase/ssr";
import { useAuth } from "@clerk/nextjs";

export function useSupabase() {
  const { getToken } = useAuth();
  const supabase = createBrowserClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      global: {
        fetch: async (input, init) => {
          const token = await getToken({ template: "supabase" }); // refresh-capable
          const headers = new Headers(init?.headers ?? {});
          if (token) headers.set("Authorization", `Bearer ${token}`);
          return fetch(input, { ...init, headers });
        }
      }
    }
  );
  return supabase;
}
```

**Behavior:** if Clerk token refresh fails, calls in flight return **401** → caught upstream by `SessionWatch` path.

```ts
// lib/net/fetch-guard.ts
export async function guardedFetch(input: RequestInfo, init?: RequestInit) {
  const res = await fetch(input, init);
  if (res.status === 401) {
    // Bubble to SessionWatch by throwing a typed error
    const body = await res.json().catch(() => ({}));
    const err: any = new Error("Session expired");
    err.code = body?.error ?? "SESSION_EXPIRED";
    throw err;
  }
  return res;
}
```

## 5) Draft Autosave (lossless UX)

**`lib/drafts/index.ts`**
```ts
export async function saveDraftSnapshot() {
  // Implement per-screen; default fallback:
  const key = `draft:${location.pathname}`;
  const payload = { ts: Date.now(), state: window.__APP_UI_STATE__ ?? null };
  localStorage.setItem(key, JSON.stringify(payload));
}

export function restoreDraftSnapshot(pathname: string) {
  const key = `draft:${pathname}`;
  const raw = localStorage.getItem(key);
  if (!raw) return null;
  try { return JSON.parse(raw); } catch { return null; }
}

export function clearDraftSnapshot(pathname: string) {
  localStorage.removeItem(`draft:${pathname}`);
}
```

**On successful re-auth:**
- In `app/sign-in/complete` (post-auth callback), read `returnTo`, `restoreDraftSnapshot(returnTo)`, rehydrate the form/state, then `clearDraftSnapshot(returnTo)`.
