```swift
let ctaButton = UIButton(type: .system)
ctaButton.setTitle("Continue →", for: .normal)
ctaButton.setTitleColor(.label, for: .normal) // dynamic black/white
ctaButton.titleLabel?.font = .systemFont(ofSize: 17, weight: .semibold)
ctaButton.translatesAutoresizingMaskIntoConstraints = false
ctaButton.layer.cornerRadius = 28
ctaButton.layer.masksToBounds = true

// Liquid Glass-style blur background
let blurEffect = UIBlurEffect(style: .systemUltraThinMaterialLight)
let blurView = UIVisualEffectView(effect: blurEffect)
blurView.isUserInteractionEnabled = false
blurView.layer.cornerRadius = 28
blurView.layer.masksToBounds = true
blurView.translatesAutoresizingMaskIntoConstraints = false
ctaButton.insertSubview(blurView, at: 0)

// Add subtle border
ctaButton.layer.borderWidth = 1
ctaButton.layer.borderColor = UIColor.white.withAlphaComponent(0.2).cgColor

// Constraint blur to button bounds
NSLayoutConstraint.activate([
    blurView.topAnchor.constraint(equalTo: ctaButton.topAnchor),
    blurView.bottomAnchor.constraint(equalTo: ctaButton.bottomAnchor),
    blurView.leadingAnchor.constraint(equalTo: ctaButton.leadingAnchor),
    blurView.trailingAnchor.constraint(equalTo: ctaButton.trailingAnchor),
])
```

view.addSubview(ctaButton)
